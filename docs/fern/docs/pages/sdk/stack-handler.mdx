# StackHandler

Handles routing and rendering for various Stack authentication and account management pages.

## Props

- `app`: `StackServerApp<HasTokenStore>` - The Stack server application instance.
- `params` (optional): `{ stack?: string[] }` - URL parameters, where `stack` is an array of path segments.
- `searchParams` (optional): `Record<string, string>` - URL search parameters.
- `fullPage` (optional): `boolean` - Whether to render the component as a full page. Default is `true` (deprecated).

## Example

```tsx
import { StackHandler } from '@stackframe/stack';

export default async function Page({
  params,
  searchParams,
}: {
  params: { stack: string[] };
  searchParams: { [key: string]: string | string[] | undefined };
}) {
  return (
    <StackHandler
      app={app}
      params={params}
      searchParams={searchParams}
      fullPage={true}
    />
  );
}
```
